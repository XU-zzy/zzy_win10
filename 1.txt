#include <iostream>
using namespace std;
int a[] = {1,3,5,7,9,2,4,6,8,10};
int ans = 0;
int n = 10;
int main()
{
    for(int i = 0; i < sizeof(a)/sizeof(int); i++) {
        for(int j = i + 1; j < sizeof(a)/sizeof(int); j++) {
            if(a[j] > a[i]) ans = max(ans,a[j] - a[i]);
        }
    }
    cout << ans << endl;
    getchar();
    return 0;
}
